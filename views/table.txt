CREATE TABLE `fly_member` (
	`memNo` INT(11) NOT NULL AUTO_INCREMENT COMMENT '회원번호',
	`memId` VARCHAR(20) NULL DEFAULT NULL COMMENT '아이디' COLLATE 'utf8_general_ci',
	`memPw` VARCHAR(65) NULL DEFAULT NULL COMMENT '비밀번호' COLLATE 'utf8_general_ci',
	`memNm` VARCHAR(20) NULL DEFAULT NULL COMMENT '회원명' COLLATE 'utf8_general_ci',
	`mobile` VARCHAR(13) NULL DEFAULT NULL COMMENT '휴대전화' COLLATE 'utf8_general_ci',
	`isAdmin` TINYINT(1) NULL DEFAULT '0' COMMENT '1 - 관리자, 0 - 회원',
	`regDt` DATETIME NULL DEFAULT current_timestamp(),
	`modDt` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`memNo`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;
